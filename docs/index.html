 💡 Time based Waits

😟 Overly relying on time based waits

it('should display a list of todos', () => {
  cy.visit('/')
  cy.wait(1000)
  cy.get('#my-todos').should('be.visible')
})

it('should load records', () => {
  cy.visit('/')
  cy.wait(5000)
  cy.get('#load-records').click()
  cy.wait(7500)
  cy.get('#my-records').should('be.visible')
})

✅ Spy using cy.intercept

it('should display a list of todos', () => {
  cy.intercept('GET', '/todos').as('todos')
  cy.visit('/')
  cy.wait('@todos')
  cy.get('#my-todos').should('be.visible')
})

✅ Spread to handle multiple requests

cy.intercept('GET', '/todos').as('todos')
cy.intercept('GET', '/users').as('users')
cy.intercept('GET', '/comments').as('comments')
cy.wait(['@todos', '@users', '@comments']).spread(
  (todos, users, comments) => {
    // each interception is now an individual argument
  }
)

      url: '/login',

      body: {
        email: loginEmail,
        password: loginPassword,
      },
    })
      .its('body')

        }
      })

 💡 Not isolating tests

describe('add todo form', () => {
  it('should visit the add todo form', () => {
    cy.visit('/todos/new')
  })

  it('should add a todo', () => {
    cy.get('.new-todo').type('first item{enter}')
  })

  it('should add a todo priority', () => {
    cy.get('.new-todo-priority').type('1{enter}')
  })

  it('should submit a valid add todo form', () => {
    cy.get('form').submit()
  })
})

✅ Combine into one test

describe('add todo form', () => {
  beforeEach(() => {
    cy.visit('/todos/new')
    cy.get('.new-todo').type('first item{enter}')
    cy.get('.new-todo-priority').type('1{enter}')
  })
  
  it('should display an error when todo priority is missing', () => {
    cy.get('.new-todo-priority').clear()
    cy.get('form').submit()
    cy.get('.form-errors').should('contain', 'Todo priority is required')
  })
  
  it('should submit a valid add todo form', () => {
    cy.get('form').submit()
  })
})

    .its('response.body')
    .should('have.length', 0)

✅ Stubs

it('starts with zero items (fixture)', () => {
  cy.intercept('GET', '/todos', { fixture: 'empty-list.json' }).as('todos')

    .its('response.body')
