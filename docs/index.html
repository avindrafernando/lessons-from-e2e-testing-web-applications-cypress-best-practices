describe('add todo form', () => {
  beforeEach(() => {
    cy.visit('/todos/new')
    cy.get('.new-todo').type('first item{enter}')
    cy.get('.new-todo-priority').type('1{enter}')
  })
  
  it('should display an error when todo priority is missing', () => {
    cy.get('.new-todo-priority').clear()
    cy.get('form').submit()
    cy.get('.form-errors').should('contain', 'Todo priority is required')
  })
  
  it('should submit a valid add todo form', () => {
    cy.get('form').submit()
  })
})
